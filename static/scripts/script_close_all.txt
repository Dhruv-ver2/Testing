import pygetwindow as gw
import time

def close_all_except_me():
    current_app = gw.getActiveWindow()
    my_title = current_app.title if current_app else None
    
    # Items never have to close
    ignore_list = ["Program Manager", "Start", "Taskbar", "Settings", "Microsoft Text Input Application"]

    print(f"Targeting all apps except: {my_title}")

    all_windows = gw.getAllWindows()
    
    # Close everything else
    for window in all_windows:
        if window.title and window.visible:
            # Skip system apps
            if any(ignore in window.title for ignore in ignore_list) or window.title == my_title:
                continue
            
            try:
                window.close()
            except:
                pass

    if current_app:
        print("Closing the script window...")
        time.sleep(1) # see it finished
        current_app.close()

if __name__ == "__main__":
    close_all_except_me()
